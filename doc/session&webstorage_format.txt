■session(cookie)
encrypted_id        # user_id, player_idを暗号化した文字列

■webstorage
・全般
  - コンペ開始時に必要な情報を読み込む。
  - コンペ終了時にはすべて削除する

・フォーマット
competition_status:
  {
    competition_id
    loaded                # hole情報をすべて読み込んだ時点でtrueとする
    holes:                # コンペ開始時にすべて読み込む
      [
        {
          hole_id 
          hole_no
          par
          yard
          order           # プレイするホールの順番(1～18)
        }
        ..
      ]
    self_score:           # ホールが終了する度にエントリを追加する。定期的にバックグラウンドでサーバに結果を送信する
      [
        {
          hole_id 
          shot_num
          pat_num
          sended          # サーバに送信済みかどうか
        }
        ..
      ]

    other_scores:         # スコア閲覧時に随時読み込んで保存。
      [
        {
          player_id
          score:
            {
              hole_id
              hole_no
              shot_num
              pat_num
            } 
            ..
        }
        ..
      ]
  }
